#include <iostream>
#include <vector>
#include <queue>
#include <algorithm>
using namespace std;

int N;
int tmp = 1;
int village;
int map[25][25];
int dx[] = { -1,1,0,0 };
int dy[] = { 0,0,-1,1 };
bool visit[25][25];
vector<int> home;
queue<pair<int, int>> q;

void bfs(int x, int y) {
	if (visit[x][y] == false && map[x][y] == 1) { // 방문한 적 있는 집인지 확인
		visit[x][y] = true; // 방문 처리
		q.push({ x,y });

		while (!q.empty()) {
			int x = q.front().first;
			int y = q.front().second;
			q.pop();

			for (int i = 0; i < 4; i++) {
				int nx = x + dx[i];
				int ny = y + dy[i];

				if (nx < N && nx >= 0 && ny < N && ny >= 0) { // 범위 내에 있는지 확인 
					if (visit[nx][ny] == false && map[nx][ny] == 1) {
						q.push({ nx,ny });
						visit[nx][ny] = true;
						tmp += 1; // 새로 방문한 집의 수 추가
					}
				}
			}
		}
		village += 1;

		if (tmp != 0) {
			home.push_back(tmp);
			tmp = 1;
		}
	}
}

int main() {
	cin >> N;
	for (int i = 0; i < N; i++)
		for (int j = 0; j < N; j++)
			scanf("%1d", &map[i][j]);

	for (int i = 0; i < N; i++)
		for (int j = 0; j < N; j++)
			bfs(i, j);

	sort(home.begin(), home.end());

	cout << village << endl;
	for (auto i = home.begin(); i != home.end(); i++)
		cout << *i << endl;
}
