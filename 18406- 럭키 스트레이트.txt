#include <iostream>
using namespace std;

int main() {
	string N;
	cin >> N;
	int len = N.length();
	int leftSum = 0;
	int rightSum = 0;

	for (int i = 0; i < len / 2; i++)
		leftSum += N[i] - '0';

	for (int i = len/2; i < len; i++)
		rightSum += N[i] - '0';

	if (leftSum == rightSum)
		cout << "LUCKY";
	else
		cout << "READY";
}
