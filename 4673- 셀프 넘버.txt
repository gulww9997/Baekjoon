#include <iostream>
#include <string>
using namespace std;

bool check[10000];

int sequence(int n) {
	string s = to_string(n);
	int result = n;
	for (int i = 0; i < s.length(); i++)
		result += (int)(s.at(i) - '0');
	return result;
}

int main() {
	for (int i = 0; i < 10000; i++)
		check[i] = true;

	int index;
	int count = 1;
	while (true) {
		if (check[count] == true)
			cout << count << endl;

		index = sequence(count);
		check[index] = false;
		count++;

		if (count == 10000)
			break;
	}
}
