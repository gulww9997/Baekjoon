#include <iostream>
using namespace std;

int map[50][50];
bool visit[50][50];
int dx[] = { -1,1,0,0 };
int dy[] = { 0,0,-1,1 };
int worm;
int M;
int N;
int K;

void dfs(int x, int y) {
	visit[x][y] = true; // 방문 처리

	for (int i = 0; i < 4; i++) { // 상하좌우 확인
		int nx = x + dx[i];
		int ny = y + dy[i];

		if (nx >= 0 && nx < N && ny >= 0 && ny < M) { // 범위 내에 존재
			if (visit[nx][ny] == true || map[nx][ny] == 0) // 방문한 적 있거나 배추가 없으면
				continue;

			if (visit[nx][ny] == false && map[nx][ny] == 1) // 방문한 적 없고 배추가 있으면
				dfs(nx, ny);
		}
	}
}

void initial() {
	for (int i = 0; i < 50; i++) // 방문 초기화
		for (int j = 0; j < 50; j++)
			visit[i][j] = false;

	for (int i = 0; i < 50; i++) // 맵 초기화
		for (int j = 0; j < 50; j++)
			map[i][j] = 0;
}

int main() {
	int T;
	cin >> T;

	for (int i = 0; i < T; i++) {
		cin >> M >> N >> K;
		for (int i = 0; i < K; i++) {
			int x, y;
			cin >> x >> y;
			map[y][x] = 1;
		}

		for (int i = 0; i < N; i++) {
			for (int j = 0; j < M; j++) {
				if (visit[i][j] == false && map[i][j] == 1) { // 방문한 적 없는 배추칸이면
					worm += 1;
					dfs(i, j);
				}
			}
		}

		cout << worm << endl;
		worm = 0;
		initial();
	}
}
