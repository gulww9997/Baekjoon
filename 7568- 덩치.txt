#include <iostream>
using namespace std;

int main() {
	int N;
	cin >> N;
	int* height = new int[N];
	int* weight = new int[N];
	int* rank = new int[N];

	for (int i = 0; i < N; i++) 
		cin >> height[i] >> weight[i];

	for (int i = 0; i < N; i++)
		rank[i] = 1;
	
	for (int i = 0; i < N; i++)
		for (int j = 0; j < N; j++)
			if (weight[j] > weight[i] && height[j] > height[i])
				rank[i]++;

	for (int i = 0; i < N; i++)
		cout << rank[i] << " ";
}
