#include <iostream>
using namespace std;

int dp[501][501];
int dpMax;

int main() {
	int n;
	cin >> n;

	for (int i = 1; i <= n; i++)
		for (int j = 1; j <= i; j++)
			scanf("%d", &dp[i][j]); // dp 테이블에 바로 초기화

	int temp = 0;
	int up, upleft = 0; // 선택하는 경우가 위, 왼쪽위 2가지임

	for (int i = 1; i <= n; i++) {
		for (int j = 1; j <= i; j++) {
			if (j == 1) // 맨 왼쪽에 있으면 선택 가능한 것은 위 밖에 없다
				upleft = 0;
			else
				upleft = dp[i - 1][j - 1];

			if (j == i) // 맨 오른쪽에 있으면 선택 가능한 것은 왼쪽위 밖에 없다
				up = 0;
			else
				up = dp[i - 1][j];

			dp[i][j] = dp[i][j] + max(up, upleft); // 현재 값 갱신
			temp = max(temp, dp[i][j]); // 최댓값 갱신
		}
		dpMax = temp;
	}
	cout << dpMax;
}
