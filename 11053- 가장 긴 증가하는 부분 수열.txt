#include <iostream>
#include <algorithm>
using namespace std;

/* 가장 긴 증가하는 부분 수열
   dp[i]는 arr[i]를 마지막 원소로 갖는 최대 수열의 길이*/

int N, result;
int arr[1001];
int dp[1001];

int main() {
	cin >> N;
	for (int i = 0; i < N; i++)
		scanf("%d", &arr[i]);

	for (int i = 0; i < N; i++)
		dp[i] = 1;

	for (int i = 0; i < N; i++) {
		for (int j = i; j < N; j++) {
			if (arr[i] < arr[j]) { // 더 큰 원소를 찾았을 때
				dp[j] = max(dp[j], dp[i] + 1);
			}
		}
		result = max(result, dp[i]);
	}

	cout << result;
}
