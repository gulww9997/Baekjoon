#include <iostream>
using namespace std;

string word;
int cnt;
bool check = true;
char tmp;

int main() {
	int N;
	cin >> N;

	for (int i = 0; i < N; i++) {
		cin >> word;

		for (int i = 0; i < word.size(); i++) {
			tmp = word[i];
			if (word[i] == word[i + 1]) {
				continue;
			}

			for (int j = i + 1; j < word.size(); j++) {
				if (tmp == word[j]) {
					check = false;
				}
			}
		}

		if (check == true)
			cnt++;
		check = true;
	}

	cout << cnt;
}
