#include <iostream>
using namespace std;

int main(){
	string s;
	cin >> s;
	int zero = 0; 
	int one = 0; 

	if (s[0] == '0')
		one++;
	else
		zero++;

	for (int i = 0; i < s.length(); i++) {
		if (s[i] != s[i + 1]) {
			if (s[i] == '0')
				one++;
			else
				zero++;
		}
	}
	cout << min(one, zero);
}
