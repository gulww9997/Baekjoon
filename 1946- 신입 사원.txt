#include <iostream>
#include <vector>
#include <algorithm>
#define INF 1000001
using namespace std;
int T, N;
int pass;
int result[20];
vector<pair<int, int>> emp[20];

bool cmp(pair<int, int> a, pair<int, int> b) {
	return a.first < b.first;
}

int main() {
	cin >> T;

	for (int i = 0; i < T; i++) { // T개의 테스트 케이스
		cin >> N;
		for (int j = 0; j < N; j++) { // 지원자 석차 입력
			int first, second;
			cin >> first >> second;
			emp[i].push_back({ first,second });
		}

		sort(emp[i].begin(), emp[i].end(), cmp); // 1차시험 석차 순으로 정렬

		int temp = INF;
		for (int j = 0; j < emp[i].size(); j++) { // 2차시험 석차로 불합 결정
			if (temp > emp[i][j].second) {
				temp = emp[i][j].second;
				pass++;
			}
		}
		result[i] = pass;
		pass = 0;
	}

	for (int i = 0; i < T; i++) {
		cout << result[i] << endl;
	}
}
