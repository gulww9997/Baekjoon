#include <iostream>
using namespace std;

int arr[1001];

int main() {
	int N, min, index, temp;
	cin >> N;
	for (int i = 0; i < N; i++)
		cin >> arr[i];

	for (int i = 0; i < N; i++) {
		min = 1001;
		for (int j = i; j < N; j++) {
			if (min > arr[j]) {
				min = arr[j];
				index = j;
			}
		}
		temp = arr[i];
		arr[i] = arr[index];
		arr[index] = temp;
	}

	for (int i = 0; i < N; i++)
		cout << arr[i] << endl;
}
