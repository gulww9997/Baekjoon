#include <iostream>
#include <vector>
#include <queue>
using namespace std;

int w, h;
int map[51][51];
bool visit[51][51];
int result;
int dx[] = { -1,1,0,0,-1,1,-1,1 };
int dy[] = { 0,0,-1,1,1,1,-1,-1 };
queue<pair<int, int>> start;
queue<pair<int, int>> q;

void island(int x, int y) {
	if (visit[x][y] == false) { // 방문하지 않았다면
		q.push({ x, y });
		visit[x][y] = true; // 방문처리

		while (!q.empty()) { // 이어진 땅 모두 방문
			int x = q.front().first;
			int y = q.front().second;
			q.pop();

			for (int i = 0; i < 8; i++) {
				int nx = x + dx[i];
				int ny = y + dy[i];

				if (ny <= w && ny >= 1 && nx <= h && nx >= 1) {
					if (map[nx][ny] == 1 && visit[nx][ny] == false) { // 방문하지 않은 땅이면
						q.push({ nx,ny });
					}
					visit[nx][ny] = true;
				}
			}
		}
		result++;
	}
}

int main() {
	while (true) {
		cin >> w >> h;

		if (w == 0 && h == 0)
			break;

		for (int i = 1; i <= h; i++)
			for (int j = 1; j <= w; j++) {
				cin >> map[i][j];

				if (map[i][j] == 1)
					start.push({ i,j });
			}

		while (!start.empty()) {
			island(start.front().first, start.front().second);
			start.pop();
		}

		cout << result << endl;

		result = 0;
		for (int i = 1; i <= h; i++) // 방문 기록 초기화
			for (int j = 0; j <= w; j++)
				visit[i][j] = false;
	}
}
