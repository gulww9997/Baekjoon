#include <iostream>
#include <queue>
#include <map>
using namespace std;
long A, B;
int result;
map<long , bool> m;
queue<pair<long, long>> q; 

void bfs() {
	q.push({ A,0 });

	while (!q.empty()) {
		long now = q.front().first;
		int result = q.front().second;
		q.pop();
		m[now] = true;

		if (now == B) {
			cout << result + 1;
			return;
		}

		if (now * 2 <= B && !m[now * 2]) {
			q.push({ now * 2,result + 1 });
			m[now * 2] = true;
		}

		if (10 * now + 1 <= B && !m[10 * now + 1]) {
			q.push({ 10 * now + 1,result + 1 });
			m[10 * now + 1] = true;
		}
	}
	cout << "-1";
}

int main() {
	cin >> A >> B;
	bfs();
}
