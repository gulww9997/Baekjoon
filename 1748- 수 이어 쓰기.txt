#include <iostream>
#include <math.h>
#include <string>
using namespace std;

int main() {
	int N;
	cin >> N;
	string s = to_string(N);
	int result = 0;

	for (int i = s.length(); i > 0; i--) {
		if (s.length() == 1)
			result += N;
		else if (i == 1 && s.length() > 1)
			result += 9;
		else {
			result += i * (N - pow(10, i - 1) + 1);
			N -= N - pow(10, i - 1) + 1;
		}
	}
	cout << result;
}
