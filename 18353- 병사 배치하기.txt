#include <iostream>
using namespace std;

/* 가장 긴 감소하는 부분 수열
   수열을 뒤집은 후, 가장 긴 증가하는 부분 수열로 푼다
 */

int N, result;
int knight[2001];
int dp[2001]; // dp[i]는 kngiht[i]가 마지막 원소인 부분 수열의 최대 길이

int main() {
	cin >> N;
	for (int i = 0; i < N; i++)
		scanf("%d", &knight[i]);

	for (int i = 0; i < N; i++)
		dp[i] = 1;

	for (int i = N - 1; i >= 0; i--) {
		for (int j = i; j >= 0; j--) {
			if (knight[i] < knight[j])
				dp[j] = max(dp[j], dp[i] + 1);
		}
	}

	for (int i = 0; i < N; i++)
		result = max(result, dp[i]);

	cout << N - result;
}
