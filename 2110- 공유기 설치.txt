#include <iostream>
#include <vector>
#include <algorithm>
using namespace std;

int N, C;
int result;
int home[2000001];

void ip() { // 파라메트릭 서치

	int start = 1; // 거리의 최소
	int end = home[N - 1] - home[0]; // 거리의 최대

	while (start <= end) {
		int mid = (start + end) / 2;
		int count = 1;
		int now = home[0];

		for (int i = 0; i < N; i++) {
			if (home[i] - now >= mid) {
				now = home[i];
				count += 1; // 설치 가능한 공유기 수 계산
			}
		}

		if (count >= C) { 
			start = mid + 1;
			result = mid; // 최적의 경우 저장
		}
		else
			end = mid - 1;
	}
}

int main() {
	cin >> N >> C;
	for (int i = 0; i < N; i++)
		scanf("%d", &home[i]);

	sort(home, home + N);
	ip();

	cout << result;
}
