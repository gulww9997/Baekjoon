#include <iostream>
#include <algorithm>
using namespace std;

int rope[100000];

int main() {
	int N;
	cin >> N;
	for (int i = 0; i < N; i++)
		cin >> rope[i];

	sort(rope, rope + N);
	int result = rope[N - 1];

	for (int i = 0; i < N; i++) {
		if (rope[i] * (N - i) > result)
			result = rope[i] * (N - i);
	}

	cout << result;
}
